SELECT DT_MASTER.CARDNO,PR_INFO.EMPNAME,
DT_MASTER.SECRETENO,DT_MASTER.PDATE,
DT_MASTER.INTIME,DT_MASTER.OUTTIME
FROM TB_DATA_MASTER DT_MASTER
INNER JOIN TB_PERSONAL_INFO PR_INFO 
ON DT_MASTER.CARDNO=PR_INFO.CARDNO
WHERE FINYEAR=2020
AND FINMONTH=:F_MONTH
AND PDATE = TO_DATE(:DATES||' 00:00:00', 'MM/DD/YYYY HH24:MI:SS')
AND INTIME=OUTTIME
